23G4bool Contains(const G4String& name, const G4String& nameList)
24{
25 // Append a space to both searched name and the list
26 // in order to exclude a match for names which are only substrings of
27 // some name present in the list.
28 // Eg. when omega_c0 is in the list and omega is checked for a presence
29
30 G4String checkName(name);
31 checkName.append(" ");
32
33 G4String checkNameList(nameList);
34 checkNameList.append(" ");
35
36 return (checkNameList.find(checkName) != std::string::npos);
37}

81void TG4StackPopperPhysics::ConstructProcess()
82{
85
86 fStackPopperProcess = new TG4StackPopper();
87
88 auto aParticleIterator = GetParticleIterator();
89 aParticleIterator->reset();
90 while ((*aParticleIterator)()) {
91
92 G4ParticleDefinition* particle = aParticleIterator->value();
93 G4ProcessManager* pmanager = particle->GetProcessManager();
94
95 // skip particles which do not have process manager
96 if (!pmanager) continue;
97
98 // add this as an option
99 if (fSelection.size() == 0 ||
100 Contains(particle->GetParticleName(), fSelection)) {
101
102 if (VerboseLevel() > 1) {
103 G4cout << "Adding StackPopper process to "
104 << particle->GetParticleName() << G4endl;
105 }
106
107 pmanager->AddProcess(fStackPopperProcess);
108 pmanager->SetProcessOrdering(fStackPopperProcess, idxPostStep);
109 }
110 }
111
112 if (VerboseLevel() > 0) {
113 G4cout << "### Stack popper physics constructed." << G4endl;
114 }
115}
