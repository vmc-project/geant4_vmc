187void TG4ProcessMapPhysics::UpdateHadronicProcessSubType(G4VProcess* process)
188{
191
192
193 auto processName = process->GetProcessName();
194
195 if (! G4StrUtil::contains(processName, "Nuclear")) return;
196
197 std::set<std::pair<G4String, G4int>> newCodes =
198 {{ "electron", fElectronNuclear },
199 { "positron", fPositronNuclear },
200 { "muon", fMuonNuclear },
201 { "mu-", fMuonNuclear },
202 { "photon", fPhotoNuclear }};
203
204 for ( auto newCode : newCodes) {
205 if ( G4StrUtil::contains(processName, newCode.first)) {
206 process->SetProcessSubType(newCode.second);
207 return;
208 }
209 }
210}

213G4bool TG4ProcessMapPhysics::UpdateOrSkipBiasingProcess(G4VProcess* process)
214{
216
217 // Do nothing if process has defined sub type
218 if (process->GetProcessSubType() > 0) return false;
219
220 auto processName = process->GetProcessName();
221
222 if (G4StrUtil::contains(processName, "biasLimiter")) {
223 process->SetProcessSubType(STEP_LIMITER);
224 return false;
225 }
226
227 if (G4StrUtil::contains(processName, "biasWrapper")) {
228 // do not update sub type and skip this process evaluation
229 return true;
230 }
231
232 return false;
233}

246void TG4ProcessMapPhysics::ConstructProcess()
247{
250
251 if (VerboseLevel() > 1) {
252 G4cout << "TG4ProcessMapPhysics::ConstructProcess: " << G4endl;
253 }
254
255 auto pMap = TG4ProcessMap::Instance();
256 G4bool success = true;
257
258 auto aParticleIterator = GetParticleIterator();
259 aParticleIterator->reset();
260 while ((*aParticleIterator)()) {
261 // skip iteration if particle does not have a process manager
262 if (!aParticleIterator->value()->GetProcessManager()) continue;
263
264 auto processVector =
265 aParticleIterator->value()->GetProcessManager()->GetProcessList();
266
267 for (size_t i = 0; i < processVector->length(); i++) {
268
269 if (UpdateOrSkipBiasingProcess((*processVector)[i])) continue;
270
271 if ((*processVector)[i]->GetProcessSubType() == fHadronInelastic) {
272 UpdateHadronicProcessSubType((*processVector)[i]);
273 }
274
275 auto newCodes = pMap->GetCodes((*processVector)[i]);
276 if (newCodes.first == kPNoProcess) {
277 G4String text = "Unknown process codes for ";
278 text += (*processVector)[i]->GetProcessName();
279 text += " process type: ";
280 text += std::to_string((*processVector)[i]->GetProcessType());
281 text += " process sub type: ";
282 text += std::to_string((*processVector)[i]->GetProcessSubType());
283 TG4Globals::Warning("TG4ProcessMapPhysics", "ConstructProcess", text);
284 success = false;
285 }
286 }
287 }
288
289 if (VerboseLevel() > 0) {
290 if (success)
291 G4cout << "### Processes mapped to VMC codes ok" << G4endl;
292 else
293 G4cout << "### Processes mapped to VMC codes: some proceesses unknown."
294 << G4endl;
295 }
